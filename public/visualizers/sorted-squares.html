brute_force.go 
func bruteForce(nums []int) []int {
  n := len(nums)
  res := make([]int, n)
  for i := 0; i < n; i++ {
    res[i] = nums[i] * nums[i]  // square each
  }
  sort.Ints(res)  // O(n log n)
  return res
}
optimal.go 
func sortedSquares(nums []int) []int {
  n := len(nums)
  res := make([]int, n)
  left, right := 0, n-1
  for i := n-1; i >= 0; i-- {
    l, r := nums[left]*nums[left], nums[right]*nums[right]
    if l > r { res[i] = l; left++
    } else { res[i] = r; right-- }
  }
  return res
}
