func isPalindrome(s string) bool {
  left, right := 0, len(s)-1
  for left < right {
    for left < right && !isAlNum(s[left])  { left++ }
    for left < right && !isAlNum(s[right]) { right-- }
    if toLower(s[left]) != toLower(s[right]) { return false }
    left++; right--
  }
  return true
}
