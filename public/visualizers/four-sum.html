brute_force.go 
func bruteForce4Sum(nums []int, t int) [][]int {
  res := [][]int{}; seen := map[string]bool{}
  sort.Ints(nums); n := len(nums)
  for i := 0; i < n-3; i++ {
    for j := i+1; j < n-2; j++ {
      for k := j+1; k < n-1; k++ {
        for l := k+1; l < n; l++ {
          if nums[i]+nums[j]+nums[k]+nums[l] == t {
            key := fmt.Sprintf("%d,%d,%d,%d",nums[i],nums[j],nums[k],nums[l])
            if !seen[key] { res = append(res, []int{...}); seen[key]=true }
          }
        }
      }
    }
  }
  return res
}
four_sum.go 
func fourSum(nums []int, target int) [][]int {
  sort.Ints(nums); res := [][]int{}; n := len(nums)
  for i := 0; i < n-3; i++ {
    if i > 0 && nums[i] == nums[i-1] { continue }
    for j := i+1; j < n-2; j++ {
      if j > i+1 && nums[j] == nums[j-1] { continue }
      left, right := j+1, n-1
      for left < right {
        sum := nums[i]+nums[j]+nums[left]+nums[right]
        if sum == target { res = append(res,[...]); /* skip dupes */; left++; right--
        } else if sum < target { left++
        } else { right-- }
      }
    }
  }
  return res
}
